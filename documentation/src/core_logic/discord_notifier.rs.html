11pub async fn notify_human(message: &str, url: &str) -> Result<String, AppError> {
12  /// Create JSON payload: this assumes Cerebras expects a field like `content`
13  let payload = json!({
14    "content": message
15  });
16
17  let client = Client::new();
18  /// this is a Future : `Future<Output = Result<Response, Error>>`
19  let notification = client
20    .post(url)
21    // use `json(&struct)` only on something implementing `serialize`, otherwise just use `.body()` and `header()`
22    .json(&payload)
23    //.header(CONTENT_TYPE, "application/json")
24    //.body(&payload)
25    .send()
26    .await
27    .map_err(|e| AppError::Notify(e.to_string()))?;
28
29    if notification.status() != StatusCode::OK {
30      return Err(AppError::Notify(notification.status().to_string()));
31    }
32
33    let body = notification
34      .text()
35      .await
36      .map_err(|e| AppError::Notify(e.to_string()))?;
37    /// if the message reaches the discord category, discord doesn't return any message but just a 204 status_code
38
39    // we log what is sent to Discord
40    write_step_cmd_debug("\nDISCORD NOTIFICATION SENT ->\n");
41    write_step_cmd_debug(&body);
42
43    Ok(body)
44
45}
