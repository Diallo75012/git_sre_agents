1//! For Production Standard We need `Secrets` to be just built at runtime when needed
2//! and 'NOT' be stored in any `CONST` or `static` var as it would have the same lifetime as the app
3//! creating some security issues and having the credential leak, or appear in traces...etc...
4use std::env;
5use serde_json::json;
6use reqwest::header::{HeaderMap, HeaderValue, AUTHORIZATION, CONTENT_TYPE};
7use crate::{
8  errors::AppError,
9  envs_manage,
10};
11
12
13/// we create a custom type error for only headers so that we know where it is in more fine grained way
14type HeadersResult<T> = std::result::Result<T, AppError>;
15
16/// funciton that would build the headers when needed 
17pub fn get_auth_headers() -> HeadersResult<HeaderMap> {
18  let api_key = match envs_manage::get_env("CEREBRAS_API_KEY") {
19    Ok(value) => {
20      println!("Api secret found but keeping it secret...! hahaha");
21      value
22    },
23    Err(e) => {
24      println!(
25        "{}",
26        AppError::EnvSecret(format!(
27          "An error occurred while trying to access env var `CEREBRAS_API_KEY`: {}",
28           e
29        ))
30      );
31      return Err(AppError::EnvSecret("Missing API key".to_string()));
32    },
33  };
34
35  let mut headers = HeaderMap::new();
36  headers.insert(
37    AUTHORIZATION,
38    HeaderValue::from_str(&format!("Bearer {}", api_key))
39      .map_err(|e| AppError::EnvSecret(format!("Invalid API key format: {}", e)))?,
40  );
41  headers.insert(
42    CONTENT_TYPE,
43    HeaderValue::from_static("application/json"),
44  );
45
46  Ok(headers)
47}
