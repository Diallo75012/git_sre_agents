128/// this will be the buffer history of messages stored and sent to an llm, so we need to limit it a certain way
129/// changed my mind, using here just a `VecDeque` as we can have `with_capacity()` and `push_back()/push_front()` methods
130#[derive(Serialize, Deserialize, Debug, Clone, Default, PartialEq, Eq)]
131pub struct MessageHistory {
132  /// so will have `MessageToAppend` and normal LlmResponse.choices[0].message.content formatted to a `MessageToAppend`
133  /// `LlmResponse.choices[0]` (doesn't change), `ResponseChoices.message` (`.message`), `ReponseMessage.content` (`.content`)
134  pub messages: VecDeque<MessageToAppend>,
135}
136
137type MessageHistoryResult<T> = std::result::Result<T, AppError>;
138impl MessageHistory {
139  // we create the function that would instantiate a new `MessageHistory`
140  // and then create another function that would be used to append `MessageToAppend`
141  pub fn new() -> Self {
142    let deque: VecDeque<MessageToAppend> = VecDeque::with_capacity(3);
143    Self {
144      messages: deque,
145    }
146  }
147  
148  pub fn append_message_to_history(&mut self, message_to_append: &MessageToAppend) -> MessageHistoryResult<serde_json::Value> {
149    let mut messages_present = self.messages.clone();
150  	if messages_present.len() >= 3 {
151  	  messages_present.pop_front();
152  	  messages_present.push_back(message_to_append.clone());
153  	  self.messages = messages_present;
154  	} else {
155  	  messages_present.push_back(message_to_append.clone());
156  	  self.messages = messages_present;  	    
157  	}
158    Ok(json!(self.clone()))
159  }
160  // clean up the history
161  pub fn clear_history(&mut self) -> MessageHistoryResult<()> {
162  	self.messages.clear();
163  	Ok(())
164  }
165}

215pub struct FunctionParametersContainer {}

217type FunctionParametersContainerResult<T> = std::result::Result<T, AppError>;
218impl FunctionParametersContainer {
219  /// We create a `.new()`` object for `property` field of the `tool object`
220  /// ```rust
221  /// let c = HashMap::from(
222  ///   [
223  ///     ("name".to_string(), "completion".to_string()),
224  ///     ("type".to_string(), "boolean".to_string()),
225  ///     ("description".to_string(), "job done or not?".to_string()),
226  ///   ]
227  /// );
228  /// ```
229  /// from that we can use the fn `create_function_parameters_object`
230  /// to get this:
231  /// `{ "completion": {"description": "job done or not?", "type": "boolean"}, {...}}`
232  pub fn create_function_parameters_object(
233    // we will use `.fn_param_as_map()` to create variabled and put those into a `Vec` to make up the `param_settings`
234    param_settings: &[HashMap<String, String>]  
235  ) -> FunctionParametersContainerResult<HashMap<String, HashMap<String, String>>> {
236    let mut outer_hashmap = HashMap::new();
237    let mut inner_hashmap = HashMap::new();
238
239    // we loop over our input parameters objects which hold all information needed
240    for elem in param_settings.iter() {
241      // we fill the `inner_hashmap`
242      for (_idx, key) in elem.iter().enumerate() {
243      	if key.0 == "type" {
244      	  inner_hashmap.insert("type".to_string(), elem[key.0].to_string());
245      	} else if key.0 == "description" {
246      	  inner_hashmap.insert("description".to_string(), elem[key.0].to_string());
247      	}
248      }
249      // we fill then the `outer_hashmap` mapping to the name of the parameter
250      for (_idx, key) in elem.iter().enumerate() {
251      	if key.0 == "name" {
252      	  outer_hashmap.insert(elem[key.0].to_string(), inner_hashmap.clone());
253      	}
254      }
255      // we clear up the `inner_hashmap` to use it again
256      inner_hashmap.clear();
257    }
258    // we return the `outer_hashmap`
259    Ok(outer_hashmap)
260  }
261}

733    // we start here with a normal `payload` basic one with text and will add some more fields if we got some tools or structured output.
734    let mut payload = json!({
735      "model": model,
736      // this field is correct in Python as it is installed dependencies but in rust we need to mimmic curl
737      // and this field is not in the documentation of Cerebras and would return to use a malformed api call error code 422. so we excluse it
738      //"provider": { "only": ["Cerebras"] },
739      "messages": messages,
740      "stream": false,
741    });
742
743    if let Some(tool_list) = tools {
744      payload["tools"] = json!(tool_list);
745    }
746    if let Some(choice) = tool_choice {
747      payload["tool_choice"] = json!(format!("{}", match choice {
748        ChoiceTool::None => "none",
749        ChoiceTool::Auto => "auto",
750        ChoiceTool::Required => "required",
751      }));
752    }
753    if let Some(format_map) = response_format {
754      payload["response_format"] = json!(format_map);
755    }
756
757    Ok(json!(payload))
758  }
